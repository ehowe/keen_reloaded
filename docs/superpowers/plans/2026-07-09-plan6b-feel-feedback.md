# Plan 6b — Feel & Feedback Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Polish the HUD into an icon-based display (hearts + ammo shots + score, plus an overworld "Levels cleared: N / M") and add default gamepad bindings to the 8 existing input actions.

**Architecture:** A new dedicated `Hud` scene (`src/ui/hud.tscn` + `hud.gd`) is instantiated by `LevelRuntime._build_hud()` (replacing the inline `Label`), with two modes (LEVEL / OVERWORLD) and setters driven by the player's existing signals. Four 16×16 placeholder icon PNGs are generated by a one-off headless tool (`tools/gen_hud_icons.gd`), committed under `assets/sprites/hud/`. Gamepad support is additive: `GameManager._ensure_input_actions()` attaches `InputEventJoypadMotion`/`InputEventJoypadButton` events to the existing actions alongside the keyboard bindings.

**Tech Stack:** Godot 4.7 (stable), GDScript, GUT (Godot Unit Test), `Image.save_png`, `InputMap`, `InputEventJoypad*`.

**Spec:** `docs/superpowers/specs/2026-07-09-plan6b-feel-feedback-design.md`

**Conventions:** GDScript uses **TAB indentation** — every code block below uses tabs. Tests extend `GutTest`. Run tests via `./tests/run_all.sh`. The Godot binary is `/Users/eugene/.local/share/mise/installs/godot/4.7-stable/Godot.app/Contents/MacOS/Godot` (the `make` targets and `$GODOT` in `tests/run_all.sh` already point at it). **Never commit unless a step says so.**

**Spec deviations (intentional, noted here):**
1. **Empty icons are dim solids, not outlines.** Spec §4 said "outline only, dim" for `heart_empty`/`ammo_empty`. The generator instead draws the *same* shape in a dim color. Simpler generator, still reads as "depleted"; placeholder-only, replaceable later.
2. **No unique-name (`%`) node access.** The `Hud` script and tests use explicit node paths (`$LevelContainer/HeartsRow`). Avoids unique-name-in-owner setup; tests use `find_child`/paths.

**Critical ordering dependencies:**
- `hud.gd` has `const HEART_FULL := preload("res://assets/sprites/hud/heart_full.png")` (and 3 more). These preloads **hard-fail if the PNGs do not exist** when the script parses. Therefore **Task 1 (assets) must fully complete** (files generated + `make import` run) **before** Task 3 creates `hud.gd`.
- Task 3 (`Hud` scene) must complete before Task 5 (`LevelRuntime` instantiates `hud.tscn`).
- Task 2 (`Player.max_health`) must complete before Task 5 (reads `p.max_health`).

```gdscript
extends SceneTree
## One-off HUD icon generator. Writes 16x16 placeholder PNGs into
## assets/sprites/hud/. Run headless:
##   godot --headless --path . --script res://tools/gen_hud_icons.gd
## Output is committed; re-run only to regenerate. CC0 / public domain.
## All output is original/programmatically drawn — free to replace later.

const SIZE := 16
const DIR := "res://assets/sprites/hud/"

const HEART := [
	"................",
	"...##....##.....",
	"..####..####....",
	".############...",
	".############...",
	".############...",
	".############...",
	"..##########....",
	"...########.....",
	"....######......",
	".....####.......",
	"......##........",
	"................",
	"................",
	"................",
	"................",
]

const BOLT := [
	"................",
	"......####......",
	".....####.......",
	"....####........",
	"...############.",
	"...##########...",
	"....######......",
	"....######......",
	"....######......",
	".....####.......",
	"....####........",
	"...####.........",
	"..####..........",
	"................",
	"................",
	"................",
]


func _init() -> void:
	DirAccess.make_dir_recursive_absolute(DIR)
	_write("heart_full.png",  HEART, Color(0.86, 0.12, 0.16))
	_write("heart_empty.png", HEART, Color(0.30, 0.12, 0.14))
	_write("ammo_full.png",   BOLT,  Color(1.0, 0.82, 0.16))
	_write("ammo_empty.png",  BOLT,  Color(0.28, 0.23, 0.10))
	print("gen_hud_icons: wrote assets/sprites/hud")
	quit()


func _write(name: String, pattern: Array, color: Color) -> void:
	var img := Image.create(SIZE, SIZE, false, Image.FORMAT_RGBA8)
	img.fill(Color(0, 0, 0, 0))
	for y in SIZE:
		var row: String = pattern[y]
		for x in SIZE:
			if row[x] == '#':
				img.set_pixel(x, y, color)
	var err := img.save_png(DIR + name)
	if err != OK:
		push_error("gen_hud_icons: cannot write %s%s (err=%d)" % [DIR, name, err])
```

```gdscript
class_name Hud
extends Control
## Heads-up display. Two modes:
##   LEVEL     — hearts (health) + ammo shots + score
##   OVERWORLD — "Levels cleared: N / M"
## Instantiated by LevelRuntime._build_hud(). Setters are called both to seed
## initial values and from the player's score/ammo/health signals.

enum Mode { LEVEL, OVERWORLD }

const HEART_FULL  := preload("res://assets/sprites/hud/heart_full.png")
const HEART_EMPTY := preload("res://assets/sprites/hud/heart_empty.png")
const AMMO_FULL   := preload("res://assets/sprites/hud/ammo_full.png")
const AMMO_EMPTY  := preload("res://assets/sprites/hud/ammo_empty.png")
const ICON_SIZE   := 16


func set_mode(m: Mode) -> void:
	$LevelContainer.visible = (m == Mode.LEVEL)
	$OverworldContainer.visible = (m == Mode.OVERWORLD)


func set_health(hp: int, max_hp: int) -> void:
	_fill_row($LevelContainer/HeartsRow, max_hp, hp, HEART_FULL, HEART_EMPTY)


func set_ammo(ammo: int, max_ammo: int) -> void:
	_fill_row($LevelContainer/AmmoRow, max_ammo, ammo, AMMO_FULL, AMMO_EMPTY)


func set_score(score: int) -> void:
	$LevelContainer/ScoreLabel.text = "Score %d" % score


func set_cleared(cleared: int, total: int) -> void:
	$OverworldContainer/ClearedLabel.text = "Levels cleared: %d / %d" % [cleared, total]


## Rebuild `row` with `slots` TextureRects; index < filled uses full_tex, else empty_tex.
func _fill_row(row: HBoxContainer, slots: int, filled: int, full_tex: Texture2D, empty_tex: Texture2D) -> void:
	for c in row.get_children():
		c.free()
	for i in slots:
		var tr := TextureRect.new()
		tr.texture = full_tex if i < filled else empty_tex
		tr.custom_minimum_size = Vector2(ICON_SIZE, ICON_SIZE)
		tr.expand_mode = TextureRect.EXPAND_IGNORE_SIZE
		tr.stretch_mode = TextureRect.STRETCH_KEEP_ASPECT_CENTERED
		row.add_child(tr)
```

```gdscript
func _ensure_input_actions() -> void:
	# Keyboard + D-pad + left-stick for movement; face buttons for actions.
	_add_action("move_left",  [_keyev(KEY_A), _btnev(JOY_BUTTON_DPAD_LEFT),  _axisev(JOY_AXIS_LEFT_X, -1.0)])
	_add_action("move_right", [_keyev(KEY_D), _btnev(JOY_BUTTON_DPAD_RIGHT), _axisev(JOY_AXIS_LEFT_X, 1.0)])
	_add_action("move_up",    [_keyev(KEY_W), _btnev(JOY_BUTTON_DPAD_UP),    _axisev(JOY_AXIS_LEFT_Y, -1.0)])
	_add_action("move_down",  [_keyev(KEY_S), _btnev(JOY_BUTTON_DPAD_DOWN),  _axisev(JOY_AXIS_LEFT_Y, 1.0)])
	_add_action("jump",     [_keyev(KEY_SPACE), _btnev(JOY_BUTTON_A)])
	_add_action("pogo",     [_keyev(KEY_P),     _btnev(JOY_BUTTON_B)])
	_add_action("shoot",    [_keyev(KEY_X),     _btnev(JOY_BUTTON_X)])
	_add_action("interact", [_keyev(KEY_UP),    _btnev(JOY_BUTTON_Y)])


## Create the action (if absent) and attach every event. Idempotent: if the
## action already exists (e.g. declared in project.godot or a prior run), it is
## left untouched so events are never duplicated.
func _add_action(action_name: String, events: Array) -> void:
	if InputMap.has_action(action_name):
		return
	InputMap.add_action(action_name)
	for ev in events:
		InputMap.action_add_event(action_name, ev)


func _keyev(keycode: int) -> InputEventKey:
	var ev := InputEventKey.new()
	ev.physical_keycode = keycode
	return ev


func _btnev(button: int) -> InputEventJoypadButton:
	var ev := InputEventJoypadButton.new()
	ev.button_index = button
	return ev


func _axisev(axis: int, value: float) -> InputEventJoypadMotion:
	var ev := InputEventJoypadMotion.new()
	ev.axis = axis
	ev.axis_value = value
	ev.deadzone = 0.5
	return ev
```
